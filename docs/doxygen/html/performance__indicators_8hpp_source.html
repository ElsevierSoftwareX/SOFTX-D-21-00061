28  inline double perfomance_indicator( graph_t* G,
29  const dim_g_t& d_g,
30  const vertex_colors_t& C,
31  const dim_a_t& d_a,
32  const edge_weights_t& W,
33  const edge_colors_t& L,
34  const vertex_ccs_t& vCC,
35  const ccs_t& CC,
36  double pixelsize ) {
37 
38  //*****************************************************************//
39  // Light absorption morphology descriptors: //
40  //*****************************************************************//
41  foo_w_abs wfoo_abs;
42  double F_ABS = wf_abs(C,d_a,wfoo_abs,pixelsize);
43  //*****************************************************************//
44  // Exciton diffusion and dissociation morphology descriptors: //
45  //*****************************************************************//
46  double Ld = 10;
47  std::pair<double,double> F_DISS = wf_diss(G,d_g,C,W,vCC,CC,Ld,BLACK,GREEN);
48  //*****************************************************************//
49  // Charge transport morphology descriptors: //
50  //*****************************************************************//
51  unsigned int n = boost::num_vertices(*G);
52  std::vector<float> distances(n);
53  compute_shortest_distance_from_sourceC_to_targetC(RED,BLACK,
54  G, d_g, C, W,
55  distances, std::string(""));
56  double black_tort_1
57  = determine_tortuosity( C, d_a, pixelsize, distances, RED, BLACK);
58 
59  for(unsigned int i=0; i<distances.size(); i++) distances[i] = 0;
60  compute_shortest_distance_from_sourceC_to_targetC(BLUE,WHITE,
61  G, d_g, C, W,
62  distances, std::string(""));
63  double white_tort_1
64  = determine_tortuosity( C, d_a, pixelsize, distances, BLUE, WHITE);
65  double F_CT = std::min (black_tort_1, white_tort_1);
66 
67 
68  return F_ABS * F_DISS.first * F_CT;
69  } // perfomance_indicator
