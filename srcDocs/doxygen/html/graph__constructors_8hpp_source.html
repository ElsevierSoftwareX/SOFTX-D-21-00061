50  std::ifstream f(name.c_str());

52  std::string str;

57  d_g.n_bulk = d_a.nx * d_a.ny * d_a.nz;
58  unsigned int n_total = d_g.n_total();
59  M.resize(n_total);
60  for(unsigned int i=0; i< d_g.n_bulk; i++){
61  if (!f) return false;
62  f >> M[i];
63  }
64  f.close();
65  return true;
66  }
